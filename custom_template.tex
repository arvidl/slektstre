% LaTeX template for smaller code font
\documentclass[11pt]{book}
\usepackage[utf8]{inputenc}
\usepackage[norsk]{babel}
\usepackage{geometry}
\usepackage{fancyvrb}
\usepackage{listings}
\usepackage{xcolor}

% Set smaller font for code blocks
\fvset{fontsize=\footnotesize}
\DefineVerbatimEnvironment{Highlighting}{Verbatim}{fontsize=\footnotesize}

% Configure listings for smaller code
\lstset{
    basicstyle=\footnotesize\ttfamily,
    breaklines=true,
    frame=single,
    numbers=left,
    numberstyle=\tiny,
    stepnumber=1,
    tabsize=2,
    showstringspaces=false,
    showtabs=false,
    captionpos=b,
    aboveskip=0pt,
    belowskip=0pt
}

% Smaller font for verbatim environments
\makeatletter
\renewenvironment{verbatim}{%
    \footnotesize
    \@verbatim
    \fvset{fontsize=\footnotesize}%
    \@verbatim
}{%
    \end@verbatim
}
\makeatother

\begin{document}
\input{content}
\end{document}
