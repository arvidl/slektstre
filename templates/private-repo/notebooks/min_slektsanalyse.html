# =============================================================================
# GRUNNLEGGENDE STATISTIKK / BASIC STATISTICS

if 'slektstre' in locals() and slektstre.persons:
    
    # Konverter til DataFrame for analyse
    persons_data = []
    for person in slektstre.persons.values():
        persons_data.append({
            'id': person.id,
            'fornavn': person.first_name,
            'etternavn': person.last_name,
            'f√∏dsels√•r': person.birth_date.year if person.birth_date else None,
            'kj√∏nn': person.gender,
            'generasjon': slektstre.get_generation(person.id)
        })
    
    df_persons = pd.DataFrame(persons_data)
    
    print("üìä GRUNNLEGGENDE STATISTIKK / BASIC STATISTICS")
    print("=" * 50)
    
    # Antall personer per generasjon
    if 'generasjon' in df_persons.columns:
        gen_stats = df_persons['generasjon'].value_counts().sort_index()
        print(f"\nüë• Personer per generasjon / Persons per generation:")
        for gen, count in gen_stats.items():
            print(f"   Generasjon {gen}: {count} personer")
    
    # Kj√∏nnsfordeling
    if 'kj√∏nn' in df_persons.columns:
        gender_stats = df_persons['kj√∏nn'].value_counts()
        print(f"\n‚ö• Kj√∏nnsfordeling / Gender distribution:")
        for gender, count in gender_stats.items():
            print(f"   {gender}: {count} personer")
    
    # F√∏dsels√•r-statistikk
    if 'f√∏dsels√•r' in df_persons.columns:
        birth_years = df_persons['f√∏dsels√•r'].dropna()
        if len(birth_years) > 0:
            print(f"\nüìÖ F√∏dsels√•r-statistikk / Birth year statistics:")
            print(f"   Tidligste: {int(birth_years.min())}")
            print(f"   Seneste: {int(birth_years.max())}")
            print(f"   Gjennomsnitt: {int(birth_years.mean())}")
    
    # Relasjonsstatistikk
    rel_types = {}
    for rel in slektstre.relationships:
        rel_type = rel.relationship_type
        rel_types[rel_type] = rel_types.get(rel_type, 0) + 1
    
    print(f"\nüîó Relasjonstyper / Relationship types:")
    for rel_type, count in rel_types.items():
        print(f"   {rel_type}: {count}")
        
else:
    print("‚ùå Ingen data lastet inn / No data loaded")

# =============================================================================
# VISUALISERING / VISUALIZATION

if 'slektstre' in locals() and slektstre.persons:
    
    # 1. Slektstre-visualisering
    print("üé® Lager slektstre-visualisering...")
    print("üé® Creating family tree visualization...")
    
    try:
        # Bruk slektstre-pakken sin visualisering
        fig = slektstre.visualize_tree()
        fig.show()
        
    except Exception as e:
        print(f"‚ö†Ô∏è Kunne ikke lage slektstre-visualisering: {e}")
        print(f"‚ö†Ô∏è Could not create family tree visualization: {e}")
    
    # 2. Generasjonsfordeling
    if 'df_persons' in locals() and 'generasjon' in df_persons.columns:
        plt.figure(figsize=(10, 6))
        gen_counts = df_persons['generasjon'].value_counts().sort_index()
        
        plt.subplot(1, 2, 1)
        gen_counts.plot(kind='bar', color='skyblue')
        plt.title('Personer per generasjon / Persons per generation')
        plt.xlabel('Generasjon / Generation')
        plt.ylabel('Antall / Count')
        plt.xticks(rotation=45)
        
        # 3. Kj√∏nnsfordeling
        if 'kj√∏nn' in df_persons.columns:
            plt.subplot(1, 2, 2)
            gender_counts = df_persons['kj√∏nn'].value_counts()
            plt.pie(gender_counts.values, labels=gender_counts.index, autopct='%1.1f%%')
            plt.title('Kj√∏nnsfordeling / Gender distribution')
        
        plt.tight_layout()
        plt.show()
    
    # 4. F√∏dsels√•r-tidslinje
    if 'df_persons' in locals() and 'f√∏dsels√•r' in df_persons.columns:
        birth_years = df_persons['f√∏dsels√•r'].dropna()
        if len(birth_years) > 0:
            plt.figure(figsize=(12, 6))
            plt.hist(birth_years, bins=20, color='lightgreen', alpha=0.7, edgecolor='black')
            plt.title('F√∏dsels√•r-fordeling / Birth year distribution')
            plt.xlabel('√Ör / Year')
            plt.ylabel('Antall personer / Number of persons')
            plt.grid(True, alpha=0.3)
            plt.show()
    
    print("‚úÖ Visualiseringer fullf√∏rt / Visualizations completed")
    
else:

# =============================================================================
# AVANSERTE ANALYSER / ADVANCED ANALYSES
# =============================================================================

if 'slektstre' in locals() and slektstre.persons:
    
    print("üîç AVANSERTE ANALYSER / ADVANCED ANALYSES")
    print("=" * 50)
    
    # 1. Finn sentrale personer (h√∏yest grad i nettverket)
    if slektstre.graph:
        degrees = dict(slektstre.graph.degree())
        if degrees:
            max_degree = max(degrees.values())
            central_persons = [pid for pid, deg in degrees.items() if deg == max_degree]
            
            print(f"\nüåü Mest sentrale personer (grad {max_degree}):")
            print(f"üåü Most central persons (degree {max_degree}):")
            for pid in central_persons:
                person = slektstre.persons.get(pid)
                if person:
                    print(f"   - {person.first_name} {person.last_name}")
    
    # 2. Finn lengste slektslinje
    try:
        longest_lineage = slektstre.find_longest_lineage()
        if longest_lineage:
            print(f"\nüìè Lengste slektslinje ({len(longest_lineage)} generasjoner):")
            print(f"üìè Longest lineage ({len(longest_lineage)} generations):")
            for i, pid in enumerate(longest_lineage):
                person = slektstre.persons.get(pid)
                if person:
                    print(f"   {i+1}. {person.first_name} {person.last_name}")
    except Exception as e:
        print(f"‚ö†Ô∏è Kunne ikke finne lengste slektslinje: {e}")
    
    # 3. Analyse av familiegrener
    try:
        branches = slektstre.identify_family_branches()
        print(f"\nüå≥ Identifiserte familiegrener: {len(branches)}")
        print(f"üå≥ Identified family branches: {len(branches)}")
        
        for i, branch in enumerate(branches[:5]):  # Vis kun f√∏rste 5
            print(f"   Gren {i+1}: {len(branch)} personer")
    except Exception as e:
        print(f"‚ö†Ô∏è Kunne ikke identifisere familiegrener: {e}")
    
    # 4. Eksport til CSV for videre analyse
    if 'df_persons' in locals():
        export_file = "../data/analyse_eksport.csv"
        df_persons.to_csv(export_file, index=False)
        print(f"\nüíæ Data eksportert til: {export_file}")
        print(f"üíæ Data exported to: {export_file}")
        print("üí° Du kan √•pne denne filen i Excel eller andre verkt√∏y")
        print("üí° You can open this file in Excel or other tools")
    
    print("\n‚úÖ Avanserte analyser fullf√∏rt / Advanced analyses completed")
    
else:
    print("‚ùå Ingen data tilgjengelig for avanserte analyser / No data available for advanced analyses")
